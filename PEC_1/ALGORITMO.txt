ALGORITHM

CONST
VOLUMEN_CAMION: INTEGER = 3500;
END CONST

VAR
FLOW:INTEGER;
TIEMPO:REAL;
END VAR

WRITESTRING("INPUT\N");
WRITESTRING("FLOW?\N");

FLOW:= READINTEGER();

TIEMPO:= INTEGERTOREAL(VOLUMEN_CAMION/FLOW);

WRITESTRING("OUTPUT\N");
WRITESTRING("WORKING TIME: ");
WRITEREAL("%2.1f",TIEMPO);


{INPUT
FLOW?
150
OUTPUT
WORKING TIME: 23.33}

END ALGORITHM
